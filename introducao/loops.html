Loops Javascript

Abra o console no devtools apertando em F12 para ver os logs. 
            
    //Loop for
    for (let contador = 1; contador <= 5; contador++) {
        console.log(contador)
    }

    //Loop while
    var contador = 1
    while (contador <= 5) {
        console.log(contador)
        contador++
    }

    //Loop do while
    var contador = 1
    do {
        console.log(contador)
        contador++
    } while (contador <= 5)
